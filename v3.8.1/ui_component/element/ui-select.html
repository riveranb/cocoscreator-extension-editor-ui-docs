    @confirm="onConfirm($event.target.value)"

methods = {
    onChange(value: string) {
        console.log('onChange', value);
    },
    onConfirm(value: string) {
        console.log('onConfirm', value);
    },
};

Vanilla JS

<ui-select id="ui-select" value="1">
    <option value="1">Lorem ipsum dolor sit amet consectetur, adipisicing elit. Sit, sint.</option>
    <option value="2">Sapiente ratione esse voluptatibus repudiandae illo numquam quas nulla id.</option>
    <option value="3">Quidem laudantium officiis eaque quas? Repudiandae iusto expedita ad totam.</option>
    <option value="4">Praesentium adipisci, sapiente distinctio ex unde tenetur doloremque mollitia iusto!</option>
    <option value="5">Debitis temporibus animi reiciendis officiis eaque. Similique eius odit illum!</option>
</ui-select>
const element = document.getElementById('#ui-select');

// addEventListener
element.addEventListener('change', (event) => {
    // changed value
    console.log(event.target.value)
});
element.addEventListener('confirm', (event) => {
    // confirm value
    console.log(event.target.value)
});
