VUE JS

<ui-node droppable="cc.Node" ref="node"
    @change="onChange($event.target.value)"
    @confirm="onConfirm($event.target.value)"
    @cancel="onCancel($event.target.value)"
></ui-node>

Vanilla JS

<ui-node droppable="cc.Node" id="node"></ui-node>
const element = document.getElementById('#node');

// addEventListener
element.addEventListener('change', (event) => {
    const nodeUuid = event.target.value;
});
element.addEventListener('change', (event) => {
    console.log('onChange', event.target.value);
});
element.addEventListener('cancel', (event) => {
    console.log('onCancel', event.target.value);
});

Get/Set

const element = document.getElementById('#node');
// set value
element.value = nodeUuid; // nodeUuid already exist

// get value
const nodeUuid = element.value;
