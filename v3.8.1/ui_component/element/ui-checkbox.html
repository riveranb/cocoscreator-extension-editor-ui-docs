VUE JS

<ui-checkbox ref="checkbox"
    @confirm="onConfirm"
    @change="onChange"
>
    click me
</ui-checkbox>
// vue object
methods = {
    onConfirm(event) {
        this.$refs.value.innerHTML = event.target.value;
    },
    onChange(event) {
        console.log('onChange', event.target.value);
    }
};

Vanilla JS

<ui-checkbox id="checkbox">click me</ui-checkbox>
const element = document.getElementById('#checkbox');

// addEventListener
element.addEventListener('confirm', (event) => {
    document.getElementById('#value').innerHTML = event.target.value;
});
element.addEventListener('change', (event) => {
    console.log('onChange', event.target.value);
});
